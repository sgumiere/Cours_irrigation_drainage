# --- Fonctions personnalisées ---
calc_ET0 <- function(Tmin, Tmax, Rs, RH, u2, alt = 0) {
  Tmoy  <- (Tmin + Tmax) / 2
  P     <- 101.3 * ((293 - 0.0065 * alt) / 293)^5.26
  gamma <- 0.000665 * P
  es    <- (eSat(Tmax) + eSat(Tmin)) / 2
  ea    <- es * RH / 100
  Delta <- 4098 * eSat(Tmoy) / (Tmoy + 237.3)^2
  Rn    <- 0.77 * Rs
  ET0   <- (0.408*Delta*(Rn) + gamma*(37/(Tmoy+273))*u2*(es - ea)) /
           (Delta + gamma*(1 + 0.34*u2))
  return(round(ET0, 2))
}

5.4 — ET0 & bilan hydrique

Penman-Monteith FAO-56, bilan hydrique, calendrier d'irrigation

▼
\[ET_0 = \frac{0.408\,\Delta\,(R_n - G) + \gamma\,\frac{C_n}{T+273}\,u_2\,(e_s - e_a)}{\Delta + \gamma\,(1 + C_d\,u_2)}\]
# Calcul ET0 Penman-Monteith FAO-56
calc_ET0 <- function(Tmin, Tmax, RH, u2, Rs, alt = 0) {
  T     <- (Tmin + Tmax) / 2
  P     <- 101.3 * ((293 - 0.0065 * alt) / 293)^5.26
  gamma <- 0.000665 * P
  es    <- (0.6108*exp(17.27*Tmax/(Tmax+237.3)) +
            0.6108*exp(17.27*Tmin/(Tmin+237.3))) / 2
  ea    <- es * RH / 100
  Delta <- 4098 * 0.6108 * exp(17.27*T/(T+237.3)) / (T+237.3)^2
  Rn    <- 0.77 * Rs
  ET0   <- (0.408*Delta*Rn + gamma*(37/(T+273))*u2*(es-ea)) /
           (Delta + gamma*(1 + 0.34*u2))
  return(ET0)
}

# Simulation de bilan hydrique journalier
n_jours <- 120   # saison de croissance (mai-août)
ET0_jour <- rnorm(n_jours, mean = 4, sd = 1.2)
pluie    <- rpois(n_jours, lambda = 2) * rbinom(n_jours, 1, 0.3)
Kc       <- approx(seq(1,n_jours,length=length(Kc_mais)),
                   Kc_mais, xout = 1:n_jours)$y

ETc      <- ET0_jour * Kc
RFU      <- 80  # Réserve Facilement Utilisable (mm)
stock    <- numeric(n_jours)
irrigation <- numeric(n_jours)
stock[1] <- RFU

for (j in 2:n_jours) {
  stock[j] <- stock[j-1] + pluie[j] - ETc[j]
  if (stock[j] < RFU * 0.3) {
    irrigation[j] <- RFU - stock[j]
    stock[j]      <- RFU
  }
  stock[j] <- min(stock[j], RFU * 1.1)
}

# Séparation de l'écoulement de base (filtre de Lyne-Hollick)
filtre_baseflow <- function(Q, alpha = 0.925) {
  n <- length(Q)
  qf <- numeric(n)  # quick flow
  qf[1] <- Q[1]
  for (i in 2:n) {
    qf[i] <- alpha * qf[i-1] + ((1+alpha)/2) * (Q[i] - Q[i-1])
    qf[i] <- max(0, min(qf[i], Q[i]))
  }
  baseflow <- Q - qf
  return(tibble(Q = Q, quickflow = qf, baseflow = baseflow))
}

# --- Modèle proie-prédateur de Lotka-Volterra ---
lotka_volterra <- function(t, state, params) {
  with(as.list(c(state, params)), {
    dN <- alpha * N - beta * N * P   # proies
    dP <- delta * N * P - gamma * P  # prédateurs
    list(c(dN, dP))
  })
}
